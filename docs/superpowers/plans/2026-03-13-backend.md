```js
const store = new Map();

function saveUser(user) {
  store.set(user.userId, user);
  return user;
}

function getUser(userId) {
  return store.get(userId) ?? null;
}

function getAllUsers() {
  return Array.from(store.values());
}

// Test helper — do not call in production code
function _reset() {
  store.clear();
}

module.exports = { saveUser, getUser, getAllUsers, _reset };
```

Create `spot-me-api/src/matcher.js`:
```js
function score(user, candidate) {
  const gymMatch = user.gym === candidate.gym ? 1 : 0;
  const sharedSlots = user.timeSlots.filter(s => candidate.timeSlots.includes(s)).length;
  const sharedFocus = user.trainingFocus.filter(f => candidate.trainingFocus.includes(f)).length;
  return gymMatch + sharedSlots + sharedFocus;
}

function getMatches(user, allUsers) {
  return allUsers
    .filter(c => c.userId !== user.userId)
    .map(c => ({ ...c, score: score(user, c) }))
    .filter(c => c.score > 0)
    .sort((a, b) => b.score - a.score);
}

module.exports = { getMatches };
```

```js
const express = require('express');
const { randomUUID } = require('crypto');
const db = require('../db');
const { getMatches } = require('../matcher');

const router = express.Router();

router.get('/health', (req, res) => {
  res.json({ status: 'ok' });
});

router.post('/profile', (req, res) => {
  const { name, gym, trainingFocus, timeSlots } = req.body;
  if (!name || !gym || !trainingFocus || !timeSlots) {
    return res.status(400).json({ error: 'name, gym, trainingFocus, and timeSlots are required' });
  }
  const user = { userId: randomUUID(), name, gym, trainingFocus, timeSlots };
  db.saveUser(user);
  res.status(201).json(user);
});

router.get('/matches/:userId', (req, res) => {
  const user = db.getUser(req.params.userId);
  if (!user) return res.status(404).json({ error: 'User not found' });
  const matches = getMatches(user, db.getAllUsers());
  res.json(matches);
});

module.exports = router;
```
